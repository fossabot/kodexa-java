001package com.kodexa.client.cloud;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.databind.ObjectMapper;
005import com.kodexa.client.KodexaException;
006import lombok.extern.slf4j.Slf4j;
007import org.apache.http.HttpResponse;
008import org.apache.http.client.config.RequestConfig;
009import org.apache.http.client.methods.HttpGet;
010import org.apache.http.client.methods.HttpPost;
011import org.apache.http.impl.client.CloseableHttpClient;
012import org.apache.http.impl.client.HttpClientBuilder;
013import org.msgpack.jackson.dataformat.MessagePackFactory;
014
015import java.io.IOException;
016
017@Slf4j
018public abstract class AbstractKodexaSession {
019
020    protected final static ObjectMapper messagePackOm;
021    protected final static ObjectMapper jsonOm;
022
023    static {
024        messagePackOm = new ObjectMapper(new MessagePackFactory());
025        messagePackOm.setSerializationInclusion(JsonInclude.Include.NON_NULL);
026
027        jsonOm = new ObjectMapper();
028        jsonOm.setSerializationInclusion(JsonInclude.Include.NON_NULL);
029    }
030
031    protected final String organizationSlug;
032    protected final String serviceSlug;
033
034
035    public AbstractKodexaSession(String organizationSlug, String serviceSlug) {
036        this.organizationSlug = organizationSlug;
037        this.serviceSlug = serviceSlug;
038    }
039
040    protected void waitForExecution(CloudSession session, CloudExecution execution) {
041        String status = execution.getStatus();
042        while (execution.getStatus().equals("PENDING") || execution.getStatus().equals("RUNNING")) {
043            try (CloseableHttpClient client = HttpClientBuilder.create()
044                    .setDefaultRequestConfig(getRequestConfig())
045                    .build()) {
046                Thread.sleep(1000);
047                String url = KodexaCloud.getUrl() + "/api/sessions/" + session.getId() + "/executions/" + execution.getId();
048                HttpGet post = new HttpGet(url);
049                post.addHeader("x-access-token", KodexaCloud.getAccessToken());
050                HttpResponse response = client.execute(post);
051                if (response.getStatusLine().getStatusCode() != 200) {
052                    throw new KodexaException("Unable to create session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
053                }
054                execution = jsonOm.readValue(response.getEntity().getContent(), CloudExecution.class);
055
056                if (!execution.getStatus().equals(status)) {
057                    log.info("Execution Status changed from " + status + " => " + execution.getStatus());
058                    status = execution.getStatus();
059                }
060            } catch (IOException | InterruptedException e) {
061                throw new KodexaException("Unable to connect to Kodexa", e);
062            }
063        }
064
065        if (execution.getStatus().equals("FAILED")) {
066            throw new KodexaException("Processing failed");
067        }
068    }
069
070    protected RequestConfig getRequestConfig() {
071        int timeout = 120; // seconds
072        return RequestConfig.custom()
073                .setConnectTimeout(timeout * 1000)
074                .setConnectionRequestTimeout(timeout * 1000)
075                .setSocketTimeout(timeout * 1000).build();
076    }
077
078    protected CloudSession createSession(CloudSessionType sessionType) {
079        log.info("Creating session in Kodexa");
080        try (CloseableHttpClient client = HttpClientBuilder.create()
081                .setDefaultRequestConfig(getRequestConfig())
082                .build()) {
083            String url = KodexaCloud.getUrl() + "/api/sessions?" + sessionType + "=" + organizationSlug + "/" + serviceSlug;
084
085            log.info("Connecting to [" + url + "]");
086
087            HttpPost post = new HttpPost(url);
088            post.addHeader("x-access-token", KodexaCloud.getAccessToken());
089            HttpResponse response = client.execute(post);
090            if (response.getStatusLine().getStatusCode() != 200) {
091                throw new KodexaException("Unable to create session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
092            }
093            CloudSession session = jsonOm.readValue(response.getEntity().getContent(), CloudSession.class);
094            log.info("Session created [" + session + "]");
095            return session;
096        } catch (IOException e) {
097            throw new KodexaException("Unable to create session on Kodexa", e);
098        }
099    }
100}
