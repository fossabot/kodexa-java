001package com.kodexa.client.cloud;
002
003import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
004import com.fasterxml.jackson.annotation.JsonInclude;
005import com.fasterxml.jackson.core.JsonProcessingException;
006import com.fasterxml.jackson.databind.ObjectMapper;
007import com.kodexa.client.Document;
008import com.kodexa.client.KodexaException;
009import lombok.Data;
010
011import java.io.IOException;
012import java.io.InputStream;
013import java.util.HashMap;
014import java.util.Map;
015import java.util.UUID;
016
017/**
018 * A service message is a request/response object for the Kodexa platform
019 */
020@Data
021@JsonIgnoreProperties(ignoreUnknown = true)
022public class ServiceMessage {
023
024    private final static ObjectMapper OBJECT_MAPPER;
025
026    static {
027        OBJECT_MAPPER = new ObjectMapper();
028        OBJECT_MAPPER.setSerializationInclusion(JsonInclude.Include.NON_NULL);
029    }
030
031    private String uuid = UUID.randomUUID().toString();
032    private String version = "1.0.0";
033
034    private ProcessingStatus status = ProcessingStatus.PENDING;
035    private ProcessingException exception = null;
036
037    private Map<String, String> metadata = new HashMap<>();
038    private Map<String, Object> options = new HashMap<>();
039    private Map<String, Map<String, Object>> stores = new HashMap<>();
040
041    private Document document;
042
043    public static ServiceMessage from(InputStream inputStream) throws IOException {
044        return OBJECT_MAPPER.readValue(inputStream, ServiceMessage.class);
045    }
046
047    public String toJson() {
048        try {
049            return OBJECT_MAPPER.writeValueAsString(this);
050        } catch (JsonProcessingException e) {
051            throw new KodexaException("Unable to serialize service message", e);
052        }
053    }
054}
