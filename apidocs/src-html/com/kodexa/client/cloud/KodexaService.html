001package com.kodexa.client.cloud;
002
003import com.kodexa.client.Document;
004import com.kodexa.client.KodexaException;
005import com.kodexa.client.pipeline.PipelineContext;
006import com.kodexa.client.registry.SourceRegistry;
007import com.kodexa.client.steps.PipelineStep;
008import lombok.extern.slf4j.Slf4j;
009import org.apache.commons.io.IOUtils;
010import org.apache.http.HttpResponse;
011import org.apache.http.client.methods.HttpPost;
012import org.apache.http.entity.ContentType;
013import org.apache.http.entity.mime.HttpMultipartMode;
014import org.apache.http.entity.mime.MultipartEntityBuilder;
015import org.apache.http.entity.mime.content.ByteArrayBody;
016import org.apache.http.entity.mime.content.StringBody;
017import org.apache.http.impl.client.CloseableHttpClient;
018import org.apache.http.impl.client.HttpClientBuilder;
019
020import java.io.IOException;
021import java.io.InputStream;
022
023/**
024 * A step that is hosted in the Kodexa Cloud
025 */
026@Slf4j
027public class KodexaService extends AbstractKodexaSession implements PipelineStep {
028
029    private final Options options;
030
031    public KodexaService(String organizationSlug, String serviceSlug) {
032        this(organizationSlug, serviceSlug, new Options());
033    }
034
035    public KodexaService(String organizationSlug, String serviceSlug, Options options) {
036        super(organizationSlug, serviceSlug);
037        this.options = options;
038    }
039
040    @Override
041    public Document process(Document document, PipelineContext context) {
042
043        CloudSession session = this.createSession(CloudSessionType.service);
044        CloudExecution execution = executeService(session, document, context);
045        waitForExecution(session, execution);
046        return document;
047    }
048
049    private CloudExecution executeService(CloudSession session, Document document, PipelineContext context) {
050        log.info("Executing service in Kodexa");
051        try (CloseableHttpClient client = HttpClientBuilder.create()
052                .setDefaultRequestConfig(getRequestConfig())
053                .build()) {
054            String url = KodexaCloud.getUrl() + "/api/sessions/" + session.getId() + "/execute";
055            log.info("Connecting to [" + url + "]");
056            HttpPost post = new HttpPost(url);
057            post.addHeader("x-access-token", KodexaCloud.getAccessToken());
058
059            MultipartEntityBuilder builder = MultipartEntityBuilder.create();
060
061            if (options.isAttachSource()) {
062                InputStream inputStream = SourceRegistry.getInstance().getSource(document);
063                ByteArrayBody fileBody = new ByteArrayBody(IOUtils.toByteArray(inputStream), "test.doc");
064                builder.setMode(HttpMultipartMode.BROWSER_COMPATIBLE);
065                builder.addPart("file", fileBody);
066            }
067
068            StringBody optionsBody = new StringBody(jsonOm.writeValueAsString(options.get()), ContentType.APPLICATION_JSON);
069            builder.addPart("options", optionsBody);
070
071            post.setEntity(builder.build());
072            HttpResponse response = client.execute(post);
073            if (response.getStatusLine().getStatusCode() != 200) {
074                throw new KodexaException("Unable to create session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
075            }
076            CloudExecution cloudExecution = jsonOm.readValue(response.getEntity().getContent(), CloudExecution.class);
077            log.info("Execution created [" + cloudExecution + "]");
078            return cloudExecution;
079        } catch (IOException e) {
080            throw new KodexaException("Unable to create session on Kodexa", e);
081        }
082    }
083
084    @Override
085    public String getName() {
086        return "Kodexa Service [" + KodexaCloud.getUrl() + "/" + organizationSlug + "/" + serviceSlug + "]";
087    }
088
089}
