001package com.kodexa.client.pipeline;
002
003import com.kodexa.client.Document;
004import com.kodexa.client.store.DataStore;
005
006import java.util.HashMap;
007import java.util.Map;
008import java.util.Set;
009
010/**
011 * The pipeline context is used to manage state through a running
012 * pipeline
013 */
014public class PipelineContext {
015
016    /**
017     * The available stores
018     */
019    public Map<String, DataStore> stores = new HashMap<String, DataStore>();
020    private Document document;
021
022    /**
023     * Get a specific store
024     *
025     * @param name the name of the store to get
026     * @return The instance of the store (or null if not found)
027     */
028    public DataStore getStore(String name) {
029        return stores.get(name);
030    }
031
032    /**
033     * Return a set of the names of the stores in the pipeline
034     *
035     * @return set of names
036     */
037    public Set<String> getStoreNames() {
038        return stores.keySet();
039    }
040
041    /**
042     * Add the given store to the pipeline context
043     *
044     * @param name  the name of the store
045     * @param store The implementation of the store
046     */
047    public void addStore(String name, DataStore store) {
048        stores.put(name, store);
049    }
050
051    /**
052     * Sets the final output document for the pipeline
053     *
054     * @param document the final output document
055     */
056    public void setOutputDocument(Document document) {
057        this.document = document;
058    }
059
060    /**
061     * Returns the final output document, or null if there isn't one
062     *
063     * @return final output document
064     */
065    public Document getOutputDocument() {
066        return this.document;
067    }
068
069}
