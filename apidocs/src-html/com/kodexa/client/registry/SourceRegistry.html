001package com.kodexa.client.registry;
002
003import com.kodexa.client.Document;
004import com.kodexa.client.KodexaException;
005import com.kodexa.client.connectors.Connector;
006
007import java.io.InputStream;
008import java.util.HashMap;
009import java.util.Map;
010
011/**
012 * The registry of source types, used in Kodexa to allow us to
013 * separate access to sources from the steps and pipeline
014 */
015public class SourceRegistry {
016
017    private static SourceRegistry INTERNAL_REGISTRY = new SourceRegistry();
018
019    private Map<String, Connector> connectors = new HashMap<>();
020
021    public static SourceRegistry getInstance() {
022        return INTERNAL_REGISTRY;
023    }
024
025    public void addConnector(Connector connector) {
026        connectors.put(connector.getName(), connector);
027    }
028
029    public InputStream getSource(Document document) {
030        String CONNECTOR_KEY = "connector";
031        if (document.getMetadata().containsKey(CONNECTOR_KEY)) {
032            String connector = String.valueOf(document.getMetadata().get(CONNECTOR_KEY));
033            if (connectors.containsKey(connector)) {
034                return connectors.get(connector).getSource(document);
035            } else {
036                throw new KodexaException("Unable to find connector [" + connector + "]");
037            }
038        } else {
039            throw new KodexaException("Document does not have connector metadata");
040        }
041    }
042}
