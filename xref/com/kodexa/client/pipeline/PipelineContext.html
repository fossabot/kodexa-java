9   /**
10   * The pipeline context is used to manage state through a running
11   * pipeline
12   */
13  public class PipelineContext {
14  
15      public Map<String, DataStore> stores = new HashMap<String, DataStore>();
16  
17      public DataStore getStore(String name) {
18          return stores.get(name);
19      }
20  
21      public Set<String> getStoreNames() {
22          return stores.keySet();
23      }
24  
25      public void addStore(String name, DataStore store) {
26          stores.put(name, store);
27      }

29  }
