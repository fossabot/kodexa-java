1   package com.kodexa.client.pipeline;

3   import com.kodexa.client.connectors.Connector;
4   import com.kodexa.client.sink.Sink;
5   import com.kodexa.client.steps.PipelineStep;
6   import lombok.extern.slf4j.Slf4j;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  /**
12   * A Pipeline allows you to put together steps from a connector to a sink
13   * to enable re-use and encapsulation of steps.
14   */
15  @Slf4j
16  public class Pipeline {
17  
18      private final Connector connector;
19      private final PipelineContext context;
20      private Sink sink;
21      private List<PipelineStep> steps = new ArrayList<>();
22  
23      public Pipeline(Connector connector) {
24          this.connector = connector;
25          this.context = new PipelineContext();
26      }
27  
28      public void addStep(PipelineStep step) {
29          steps.add(step);
30      }
31  
32      public void setSink(Sink sink) {
33          this.sink = sink;
34      }
35  
36      public PipelineContext run() {
37  
38          log.info("Starting pipeline");
39  
40          connector.forEachRemaining(document -> {
41              for (PipelineStep step : steps) {
42                  log.info("Starting step " + step.getName());
43                  long startTime = System.currentTimeMillis();
44                  document = step.process(document, context);
45                  long endTime = System.currentTimeMillis();
46                  log.info("Step processed in " + (endTime - startTime) + " ms");
47              }
48  
49              if (sink != null) {
50                  log.info("Writing to sink " + sink.getName());
51                  sink.sink(document);
52              }
53          });
54  
55          log.info("Pipeline completed");
56          return context;
57  
58      }
59  }
