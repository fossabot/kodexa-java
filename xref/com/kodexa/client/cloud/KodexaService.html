49      private CloudExecution executeService(CloudSession session, Document document, PipelineContext context) {
50          log.info("Executing service in Kodexa");
51          try (CloseableHttpClient client = HttpClientBuilder.create()
52                  .setDefaultRequestConfig(getRequestConfig())
53                  .build()) {
54              String url = KodexaCloud.getUrl() + "/api/sessions/" + session.getId() + "/execute";
55              log.info("Connecting to [" + url + "]");
56              HttpPost post = new HttpPost(url);
57              post.addHeader("x-access-token", KodexaCloud.getAccessToken());
58  
59              MultipartEntityBuilder builder = MultipartEntityBuilder.create();
60  
61              if (options.isAttachSource()) {
62                  InputStream inputStream = SourceRegistry.getInstance().getSource(document);
63                  ByteArrayBody fileBody = new ByteArrayBody(IOUtils.toByteArray(inputStream), "test.doc");
64                  builder.setMode(HttpMultipartMode.BROWSER_COMPATIBLE);
65                  builder.addPart("file", fileBody);
66              }
67  
68              StringBody optionsBody = new StringBody(jsonOm.writeValueAsString(options.get()), ContentType.APPLICATION_JSON);
69              builder.addPart("options", optionsBody);
70  
71              post.setEntity(builder.build());
72              HttpResponse response = client.execute(post);
73              if (response.getStatusLine().getStatusCode() != 200) {
74                  throw new KodexaException("Unable to create session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
75              }
76              CloudExecution cloudExecution = jsonOm.readValue(response.getEntity().getContent(), CloudExecution.class);
77              log.info("Execution created [" + cloudExecution + "]");
78              return cloudExecution;
79          } catch (IOException e) {
80              throw new KodexaException("Unable to create session on Kodexa", e);
81          }
82      }
