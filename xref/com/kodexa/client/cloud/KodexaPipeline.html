68      /**
69       * Execute the service in Kodexa
70       *
71       * @param session  The session to use
72       * @param document The document to send
73       * @param context  The context for the pipeline
74       * @return An instance of a cloud execution
75       */
76      private CloudExecution executeService(CloudSession session, Document document, PipelineContext context) {
77          log.info("Executing service in Kodexa");
78          try (CloseableHttpClient client = HttpClientBuilder.create()
79                  .setDefaultRequestConfig(getRequestConfig())
80                  .build()) {
81              String url = KodexaCloud.getUrl() + "/api/sessions/" + session.getId() + "/execute";
82              log.info("Connecting to [" + url + "]");
83              HttpPost post = new HttpPost(url);
84              post.addHeader("x-access-token", KodexaCloud.getAccessToken());
85  
86              MultipartEntityBuilder builder = MultipartEntityBuilder.create();
87  
88              if (options.isAttachSource()) {
89                  InputStream inputStream = SourceRegistry.getInstance().getSource(document);
90                  ByteArrayBody fileBody = new ByteArrayBody(IOUtils.toByteArray(inputStream), "test.doc");
91                  builder.setMode(HttpMultipartMode.BROWSER_COMPATIBLE);
92                  builder.addPart("file", fileBody);
93              }
94  
95              StringBody optionsBody = new StringBody(jsonOm.writeValueAsString(options.get()), ContentType.APPLICATION_JSON);
96              builder.addPart("options", optionsBody);
97  
98              post.setEntity(builder.build());
99              HttpResponse response = client.execute(post);
100             if (response.getStatusLine().getStatusCode() != 200) {
101                 throw new KodexaException("Unable to create session on Kodexa, check your access token and URL [" + response.getStatusLine().getStatusCode() + "]");
102             }
103             CloudExecution cloudExecution = jsonOm.readValue(response.getEntity().getContent(), CloudExecution.class);
104             log.info("Execution created [" + cloudExecution + "]");
105             return cloudExecution;
106         } catch (IOException e) {
107             throw new KodexaException("Unable to create session on Kodexa", e);
108         }
109     }
