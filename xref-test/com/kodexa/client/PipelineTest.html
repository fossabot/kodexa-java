58      public void pipelineToLayout() throws IOException {
59          KodexaCloud.setUrl("http://localhost:8080");
60          SourceRegistry.getInstance().addConnector(new FolderConnector());
61  
62          InMemorySink sink = new InMemorySink();
63          MsgPackDocumentStore msgPackDocumentStore = new MsgPackDocumentStore("/tmp/msgpack.backup");
64          Pipeline pipeline = new Pipeline(msgPackDocumentStore);
65          pipeline.addStep(new KodexaService("kodexa", "node-tagger",
66                  Options.start()
67                          .set("type_re", "line")
68                          .set("content_re", "^(F.?itchConnect|feeds)$|^Technical.*3\\.1.$")
69                          .set("tag_name", "page_header")));
70  
71          pipeline.addStep(new KodexaService("kodexa", "pattern-table-tag",
72                  Options.start()
73                          .set("table_tag_name", "Chapter9")
74                          .set("page_start_re", "^Chapter 9.*Layout$")
75                          .set("page_end_re", "^Section 11.*Files$")
76                          .set("table_header_re", "^Name Definition .* Lookup$")
77                          .set("table_end_re", "^\\d+$")
78                          .set("tags_to_ignore_re", "page_(header|footer)|ignore")
79                          .set("col_space_multiplier", 2.0)));
80  
81          pipeline.addStep(new KodexaService("kodexa", "tagged-table-extract",
82                  Options.start()
83                          .set("store_name", "chapter9")
84                          .set("table_tag_name", "Chapter9")
85                          .set("table_header_re", "^Name Definition .* Lookup$")
86                          .set("header_lines_count", 2)
87                          .set("first_col_has_text", true)
88                          .set("tables_in_page_count", 1)));
89  
90          pipeline.setSink(sink);
91          PipelineContext context = pipeline.run();
92  
93          Assert.assertTrue(msgPackDocumentStore.getCount() == 1);
94          Assert.assertTrue(context.getStoreNames().size() == 1);
95          Assert.assertTrue(context.getStore("chapter9") != null);
96          Assert.assertTrue(context.getStore("chapter9") instanceof TableStore);
97          Assert.assertTrue(((TableStore) context.getStore("chapter9")).getRows().size() == 110);
