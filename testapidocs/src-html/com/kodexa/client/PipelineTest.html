056    @Ignore
057    @Test
058    public void pipelineToLayout() throws IOException {
059        KodexaCloud.setUrl("http://localhost:8080");
060        SourceRegistry.getInstance().addConnector(new FolderConnector());
061
062        InMemorySink sink = new InMemorySink();
063        MsgPackDocumentStore msgPackDocumentStore = new MsgPackDocumentStore("/tmp/msgpack.backup");
064        Pipeline pipeline = new Pipeline(msgPackDocumentStore);
065        pipeline.addStep(new KodexaService("kodexa", "node-tagger",
066                Options.start()
067                        .set("type_re", "line")
068                        .set("content_re", "^(F.?itchConnect|feeds)$|^Technical.*3\\.1.$")
069                        .set("tag_name", "page_header")));
070
071        pipeline.addStep(new KodexaService("kodexa", "pattern-table-tag",
072                Options.start()
073                        .set("table_tag_name", "Chapter9")
074                        .set("page_start_re", "^Chapter 9.*Layout$")
075                        .set("page_end_re", "^Section 11.*Files$")
076                        .set("table_header_re", "^Name Definition .* Lookup$")
077                        .set("table_end_re", "^\\d+$")
078                        .set("tags_to_ignore_re", "page_(header|footer)|ignore")
079                        .set("col_space_multiplier", 2.0)));
080
081        pipeline.addStep(new KodexaService("kodexa", "tagged-table-extract",
082                Options.start()
083                        .set("store_name", "chapter9")
084                        .set("table_tag_name", "Chapter9")
085                        .set("table_header_re", "^Name Definition .* Lookup$")
086                        .set("header_lines_count", 2)
087                        .set("first_col_has_text", true)
088                        .set("tables_in_page_count", 1)));
089
090        pipeline.setSink(sink);
091        PipelineContext context = pipeline.run();
092
093        Assert.assertTrue(msgPackDocumentStore.getCount() == 1);
094        Assert.assertTrue(context.getStoreNames().size() == 1);
095        Assert.assertTrue(context.getStore("chapter9") != null);
096        Assert.assertTrue(context.getStore("chapter9") instanceof TableStore);
097        Assert.assertTrue(((TableStore) context.getStore("chapter9")).getRows().size() == 110);
098
099//        Document document = msgPackDocumentStore.getDocument(0);

101    }
